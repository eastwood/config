#+BEGIN_SRC 
$ ./autogen.sh

  LDFLAGS="-L/usr/local/opt/libgccjit/lib/gcc/10" \
  LD_LIBRARY_PATH="/usr/local/opt/libgccjit/lib/gcc/10" \
  ./configure --with-nativecomp
